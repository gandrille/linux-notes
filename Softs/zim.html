    Supprimer les Creation-Date

    find . -name "*.txt" -print0 | xargs -0 -n 1 sed -i "/^Creation-Date: /d"

    Corriger les espaces insécables en français

    find . -name "*.txt" -type f -print0 | xargs -0 -n 1 sed -i "s/ »/\xC2\xA0»/g"

find . -name "*.txt" -type f -print0 | xargs -0 -n 1 sed -i "s/ \!/\xC2\xA0\!/g"
